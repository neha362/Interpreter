BNF: 
program := PROGRAM variable SEMICOLON declaration_list statement_list DOT
compound statement := BEGIN statement_list END
declaration_list := VAR (declaration SEMICOLON)*
declaration := variable (COMMMA variable)* COLON type
type := INTEGER | REAL | CHAR | STRING | FUNCTION
statement_list := statement | statement SEMICOLON statement_list
statement := compound_statement | assignment_statement | comment | function | empty
comment := LBRACE (CHAR + SPACE)* RBRACE
function := variable OPAREN factor (COMMA factor)* CPAREN
assignment_statement := variable ASSIGNEQ expr
empty := 
expr := term (ADDOP expr)*
term := factor (MULOP term)*
mulop := * | / | DIV
addop := + | -
factor := ADDOP? (OPAREN expr CPAREN | number (^ factor)* | variable)
number :=  INTEGER* (PERIOD INTEGER*)?
variable := ID
